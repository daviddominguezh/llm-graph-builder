> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Architecture:** New `app_type` column on `agents` and `agent_templates` tables. Agent-specific columns (`system_prompt`, `max_steps`) on `agents` table, plus `agent_context_items` table. Wizard step 1 gains type selector cards that filter the template grid. Template sync and clone extended for both types. User-facing text renamed from "agent" to "app".

**Tech Stack:** Supabase (Postgres + RLS), Express backend, Next.js 16, React, shadcn/ui, next-intl, Zod

## Task 5a: Backend — assembleAgentTemplateConfig with MCP Stripping

**Files:**
- Modify: `packages/backend/src/db/queries/templateSync.ts`

- [ ] **Step 1: Create stripMcpServerRowsToTemplate helper**

The `stripMcpServerRows` function does not exist in the codebase. The existing `assembleTemplateSafeGraph.ts` strips MCP servers from the JSONB `graph_data` format (which has a nested `transport` object). For agent-type sync we need a different function that maps flat DB rows from `graph_mcp_servers` (columns: `server_id`, `name`, `transport_type`, `transport_config`, `library_item_id`) to `TemplateMcpServer[]`.

Create this function in `templateSync.ts`:

```ts
import type { TemplateMcpServer } from '@daviddh/graph-types';

interface McpServerDbRow {
  server_id: string;
  name: string;
  transport_type: string;
  transport_config: Record<string, unknown>;
  library_item_id: string | null;
}

function isMcpServerDbRow(value: unknown): value is McpServerDbRow {
  return typeof value === 'object' && value !== null && 'server_id' in value && 'name' in value;
}

function extractHeaderKeysFromConfig(config: Record<string, unknown>): string[] {
  const headers = config.headers;
  if (typeof headers === 'object' && headers !== null) return Object.keys(headers);
  return [];
}

function mapDbRowToTemplate(row: McpServerDbRow): TemplateMcpServer {
  if (row.library_item_id !== null) {
    return { type: 'library' as const, libraryItemId: row.library_item_id, name: row.name };
  }
  const url = typeof row.transport_config.url === 'string' ? row.transport_config.url : undefined;
  return {
    type: 'custom' as const,
    name: row.name,
    transportType: row.transport_type,
    url,
    headerKeys: extractHeaderKeysFromConfig(row.transport_config),
  };
}

function stripMcpServerRowsToTemplate(rows: unknown): TemplateMcpServer[] {
  if (!Array.isArray(rows)) return [];
  return rows.filter(isMcpServerDbRow).map(mapDbRowToTemplate);
}
```

- [ ] **Step 2: Create assembleAgentTemplateConfig function**

Create a new exported function in `templateSync.ts`:

```ts
export async function assembleAgentTemplateConfig(
  supabase: SupabaseClient,
  agentId: string
): Promise<Record<string, unknown> | null> {
  // Fetch agent's system_prompt, max_steps
  const { data: agent } = await supabase
    .from('agents')
    .select('system_prompt, max_steps')
    .eq('id', agentId)
    .single();

  if (agent === null) return null;

  // Fetch context items
  const { data: contextRows } = await supabase
    .from('agent_context_items')
    .select('content')
    .eq('agent_id', agentId)
    .order('sort_order', { ascending: true });

  const contextItems = Array.isArray(contextRows)
    ? contextRows.map((r: { content: string }) => r.content)
    : [];

  // Fetch MCP servers from graph_mcp_servers table (flat DB rows)
  const { data: mcpRows } = await supabase
    .from('graph_mcp_servers')
    .select('server_id, name, transport_type, transport_config, library_item_id')
    .eq('agent_id', agentId);

  const mcpServers = stripMcpServerRowsToTemplate(mcpRows);

  return {
    systemPrompt: agent.system_prompt ?? '',
    contextItems,
    maxSteps: agent.max_steps ?? null,
    mcpServers,
  };
}
```

const CLONE_COLUMNS = 'id, agent_id, app_type, template_graph_data, template_agent_config';

export async function getTemplateForClone(
  supabase: SupabaseClient,
  agentId: string
): Promise<{
  result: { id: string; app_type: string; template_graph_data: unknown; template_agent_config: unknown } | null;
  error: string | null;
}> {
  const { data, error } = await supabase
    .from('agent_templates')
    .select(CLONE_COLUMNS)
    .eq('agent_id', agentId)
    .maybeSingle();

  if (error !== null) return { result: null, error: error.message };
  if (data === null) return { result: null, error: null };
  return { result: data as { id: string; app_type: string; template_graph_data: unknown; template_agent_config: unknown }, error: null };
}

```ts
// packages/backend/src/db/queries/cloneAgentConfig.ts
import type { TemplateMcpServer } from '@daviddh/graph-types';

import { insertMcpServers } from './cloneTemplateGraph.js';
import type { SupabaseClient } from './operationHelpers.js';
import { throwOnMutationError } from './operationHelpers.js';

interface AgentTemplateConfig {
  systemPrompt: string;
  contextItems: string[];
  maxSteps: number | null;
  mcpServers: TemplateMcpServer[];
}

function isAgentTemplateConfig(val: unknown): val is AgentTemplateConfig {
  return typeof val === 'object' && val !== null && 'systemPrompt' in val;
}

function parseTemplateMcpServers(raw: unknown): TemplateMcpServer[] {
  if (!Array.isArray(raw)) return [];
  // Each element is already a TemplateMcpServer shape from the JSON config
  return raw as TemplateMcpServer[];
}

async function setAgentFields(
  supabase: SupabaseClient,
  agentId: string,
  config: AgentTemplateConfig
): Promise<void> {
  const result = await supabase
    .from('agents')
    .update({ system_prompt: config.systemPrompt, max_steps: config.maxSteps })
    .eq('id', agentId);
  throwOnMutationError(result, 'cloneAgentConfig:fields');
}

async function insertContextItems(
  supabase: SupabaseClient,
  agentId: string,
  items: string[]
): Promise<void> {
  if (items.length === 0) return;
  const rows = items.map((content, i) => ({
    agent_id: agentId,
    sort_order: i,
    content,
  }));
  const result = await supabase.from('agent_context_items').insert(rows);
  throwOnMutationError(result, 'cloneAgentConfig:contextItems');
}

export async function cloneAgentConfig(
  supabase: SupabaseClient,
  agentId: string,
  rawConfig: unknown
): Promise<void> {
  if (!isAgentTemplateConfig(rawConfig)) return;
  await setAgentFields(supabase, agentId, rawConfig);
  await insertContextItems(supabase, agentId, rawConfig.contextItems);

  // Clone MCP servers by parsing template_agent_config.mcpServers as TemplateMcpServer[]
  // and inserting via the same insertMcpServers from cloneTemplateGraph.ts
  const mcpServers = parseTemplateMcpServers(rawConfig.mcpServers);
  await insertMcpServers(supabase, agentId, mcpServers);
}
```
