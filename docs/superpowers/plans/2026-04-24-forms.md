# Forms Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the "forms" feature — agent-scoped structured data collection with tools, Data-tab configuration UI, right-panel inspection, streaming CSV export, and schema-coupling warnings.

**Architecture:** A new `graph_forms` table per-agent; form data stored under `conversations.metadata.forms[<formUuid>]`. Two LLM tools (`set_form_fields`, `get_form_field`) registered under `OpenFlow/Forms`, built by a pure lib in `packages/api/src/lib/forms/` that is also reused by the right panel and the CSV Route Handler. UI built with shadcn/ui following existing patterns in `packages/web`.

**Tech Stack:** TypeScript (strict, ESM), Postgres + Supabase (JSONB, RLS, triggers), Next.js 16 App Router (Server Actions + Route Handlers), Jest (ESM), shadcn/ui, next-intl.

**Spec:** [`docs/superpowers/specs/2026-04-24-forms-design.md`](../specs/2026-04-24-forms-design.md)

  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  agent_id    uuid NOT NULL REFERENCES public.agents(id) ON DELETE CASCADE,
  form_slug   text NOT NULL,
  schema_id   text NOT NULL,
  validations jsonb NOT NULL DEFAULT '{}'::jsonb,
  created_at  timestamptz NOT NULL DEFAULT now(),
  updated_at  timestamptz NOT NULL DEFAULT now(),

  CONSTRAINT graph_forms_agent_slug_unique UNIQUE (agent_id, form_slug),
  FOREIGN KEY (agent_id, schema_id)
    REFERENCES public.graph_output_schemas(agent_id, schema_id)
    ON DELETE RESTRICT,
  CONSTRAINT form_slug_format
    CHECK (form_slug ~ '^[a-z0-9]+(-[a-z0-9]+)*$'),
  CONSTRAINT form_slug_length
    CHECK (char_length(form_slug) BETWEEN 1 AND 64),
  CONSTRAINT form_slug_not_reserved
    CHECK (form_slug NOT IN ('new','all','any','none','edit','delete','create',
                             'export','import','settings','admin','api','null','undefined'))
);

CREATE INDEX graph_forms_schema_idx
  ON public.graph_forms (agent_id, schema_id);

CREATE TRIGGER graph_forms_set_updated_at
  BEFORE UPDATE ON public.graph_forms
  FOR EACH ROW EXECUTE FUNCTION public.update_updated_at();

ALTER TABLE public.graph_forms ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Users can read forms on their agents"
  ON public.graph_forms FOR SELECT
  USING (EXISTS (SELECT 1 FROM public.agents a WHERE a.id = graph_forms.agent_id AND a.user_id = auth.uid()));

// packages/api/src/lib/forms/slugNormalize.ts
const MAX_LEN = 64;

export function slugNormalize(input: string): string {
  const lower = input.toLowerCase();
  const hyphenated = lower.replace(/[\s_]+/g, '-');
  const stripped = hyphenated.replace(/[^a-z0-9-]/g, '');
  const collapsed = stripped.replace(/-+/g, '-');
  const trimmed = collapsed.replace(/^-+|-+$/g, '');
  return trimmed.slice(0, MAX_LEN);
}
```

// packages/api/src/lib/forms/parsePath.ts
import type { ParsedPath, ParseError, PathSegment } from '../../types/forms.js';

type ParseOk = { ok: true; segments: ParsedPath };
type ParseFail = { ok: false; error: ParseError };
export type ParsePathResult = ParseOk | ParseFail;

const FIELD_NAME_RE = /^[a-zA-Z_][a-zA-Z0-9_]*$/;

export function parsePath(input: string): ParsePathResult {
  if (input.length === 0) return fail('Empty path', 0);
  const rawSegments = input.split('.');
  const segments: PathSegment[] = [];
  let cursor = 0;
  for (const raw of rawSegments) {
    if (raw.length === 0) return fail('Empty segment (trailing or leading dot)', cursor);
    const parsed = parseSegment(raw, cursor);
    if (!parsed.ok) return parsed;
    segments.push(parsed.segment);
    cursor += raw.length + 1;
  }
  return { ok: true, segments };
}

type SegmentResult = { ok: true; segment: PathSegment } | ParseFail;

function parseSegment(raw: string, cursorBase: number): SegmentResult {
  const bracketStart = raw.indexOf('[');
  const name = bracketStart === -1 ? raw : raw.slice(0, bracketStart);
  if (!FIELD_NAME_RE.test(name)) return fail(`Invalid field name "${name}"`, cursorBase);
  const indicesPart = bracketStart === -1 ? '' : raw.slice(bracketStart);
  const indices = parseIndices(indicesPart, cursorBase + (bracketStart === -1 ? 0 : bracketStart));
  if (!indices.ok) return indices;
  return { ok: true, segment: { fieldName: name, indices: indices.list } };
}

type IndicesResult = { ok: true; list: Array<number | 'wildcard'> } | ParseFail;

function parseIndices(s: string, cursor: number): IndicesResult {
  const list: Array<number | 'wildcard'> = [];
  let i = 0;
  while (i < s.length) {
    if (s[i] !== '[') return fail(`Expected '[' at ${String(cursor + i)}`, cursor + i);
    const close = s.indexOf(']', i);
    if (close === -1) return fail('Missing closing bracket', cursor + i);
    const token = s.slice(i + 1, close);
    if (token === '') list.push('wildcard');
    else if (/^[0-9]+$/.test(token)) list.push(parseInt(token, 10));
    else return fail(`Invalid index "${token}"`, cursor + i + 1);
    i = close + 1;
  }
  return { ok: true, list };
}

function fail(reason: string, at: number): ParseFail {
  return { ok: false, error: { reason, at } };
}
```

// packages/api/src/lib/forms/normalizePath.ts
import { parsePath } from './parsePath.js';

export function normalizePath(runtimePath: string): string | null {
  const parsed = parsePath(runtimePath);
  if (!parsed.ok) return null;
  return parsed.segments
    .map(({ fieldName, indices }) => {
      const brackets = indices.map(() => '[]').join('');
      return `${fieldName}${brackets}`;
    })
    .join('.');
}
```

// packages/api/src/lib/forms/collectFieldPaths.ts
import type { OutputSchemaField } from '@daviddh/graph-types';

export type SimpleLeafType = 'string' | 'number';

export interface SimpleLeaf {
  path: string;
  type: SimpleLeafType;
}

export function collectFieldPaths(fields: OutputSchemaField[], prefix = ''): string[] {
  const out: string[] = [];
  for (const f of fields) collectOne(f, prefix, out);
  return out;
}

function collectOne(field: OutputSchemaField, prefix: string, out: string[]): void {
  const path = prefix === '' ? field.name : `${prefix}.${field.name}`;
  if (field.type === 'object' && field.properties) {
    for (const p of field.properties) collectOne(p, path, out);
    return;
  }
  if (field.type === 'array' && field.items) {
    collectOne(field.items, `${path}[]`, out);
    return;
  }
  out.push(path);
}

export function collectSimpleLeafPaths(fields: OutputSchemaField[]): SimpleLeaf[] {
  const all = collectFieldPaths(fields);
  const typeMap = buildTypeMap(fields);
  const result: SimpleLeaf[] = [];
  for (const path of all) {
    const t = typeMap.get(path);
    if (t === 'string' || t === 'number') result.push({ path, type: t });
  }
  return result;
}

function buildTypeMap(fields: OutputSchemaField[]): Map<string, string> {
  const map = new Map<string, string>();
  walkForTypes(fields, '', map);
  return map;
}

function walkForTypes(fields: OutputSchemaField[], prefix: string, map: Map<string, string>): void {
  for (const f of fields) {
    const path = prefix === '' ? f.name : `${prefix}.${f.name}`;
    if (f.type === 'object' && f.properties) walkForTypes(f.properties, path, map);
    else if (f.type === 'array' && f.items) walkForTypesOne(f.items, `${path}[]`, map);
    else map.set(path, f.type);
  }
}

function walkForTypesOne(field: OutputSchemaField, path: string, map: Map<string, string>): void {
  if (field.type === 'object' && field.properties) walkForTypes(field.properties, path, map);
  else if (field.type === 'array' && field.items) walkForTypesOne(field.items, `${path}[]`, map);
  else map.set(path, field.type);
}
```

// packages/api/src/lib/forms/zodForFieldPath.ts
import type { OutputSchemaField } from '@daviddh/graph-types';
import { z, type ZodTypeAny } from 'zod';

import { parsePath } from './parsePath.js';

export type ZodLookup =
  | { ok: true; zod: ZodTypeAny; expectedType: string }
  | { ok: false; reason: 'path-not-found' | 'parse-error'; detail?: string };

export function zodForFieldPath(fields: OutputSchemaField[], path: string): ZodLookup {
  const parsed = parsePath(path);
  if (!parsed.ok) return { ok: false, reason: 'parse-error', detail: parsed.error.reason };
  const field = resolveField(fields, parsed.segments);
  if (!field) return { ok: false, reason: 'path-not-found' };
  return { ok: true, zod: zodForField(field), expectedType: describeType(field) };
}

function resolveField(fields: OutputSchemaField[], segments: ReturnType<typeof parsePath> extends { ok: true; segments: infer S } ? S : never): OutputSchemaField | undefined {
  let current: OutputSchemaField | undefined;
  let pool: OutputSchemaField[] | undefined = fields;
  for (const seg of segments) {
    if (!pool) return undefined;
    current = pool.find((f) => f.name === seg.fieldName);
    if (!current) return undefined;
    let walker = current;
    for (let i = 0; i < seg.indices.length; i += 1) {
      if (walker.type !== 'array' || !walker.items) return undefined;
      walker = walker.items;
    }
    current = walker;
    pool = current.type === 'object' ? current.properties : undefined;
  }
  return current;
}

function zodForField(field: OutputSchemaField): ZodTypeAny {
  switch (field.type) {
    case 'string': return z.string();
    case 'number': return z.number();
    case 'boolean': return z.boolean();
    case 'enum': return z.enum((field.enumValues ?? ['']) as [string, ...string[]]);
    case 'array': return z.array(field.items ? zodForField(field.items) : z.unknown());
    case 'object': {
      const shape: Record<string, ZodTypeAny> = {};
      for (const p of field.properties ?? []) shape[p.name] = zodForField(p);
      return z.object(shape);
    }
  }
}

function describeType(field: OutputSchemaField): string {
  if (field.type === 'enum') return `enum(${(field.enumValues ?? []).join('|')})`;
  if (field.type === 'array' && field.items) return `array of ${describeType(field.items)}`;
  return field.type;
}

// packages/api/src/lib/forms/runValidation.ts
import type { ValidationRule } from '../../types/forms.js';

export type ValidationOutcome = { ok: true } | { ok: false; reason: string };

const EMAIL_RE = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
const DATE_RE = /^\d{4}-\d{2}-\d{2}$/;
const HOUR_RE = /^([01]\d|2[0-3]):[0-5]\d$/;

export function runValidation(value: unknown, rule: ValidationRule): ValidationOutcome {
  switch (rule.kind) {
    case 'email': return checkString(value, (s) => EMAIL_RE.test(s), 'Expected a valid email');
    case 'twoWordName': return checkString(value, isTwoWordName, 'Expected two words, each ≥2 chars');
    case 'pastDate': return checkDate(value, 'past');
    case 'futureDate': return checkDate(value, 'future');
    case 'pastHour': return checkHour(value, 'past');
    case 'futureHour': return checkHour(value, 'future');
    case 'length': return checkLength(value, rule);
  }
}

function checkString(v: unknown, predicate: (s: string) => boolean, reason: string): ValidationOutcome {
  if (typeof v !== 'string' || !predicate(v)) return { ok: false, reason };
  return { ok: true };
}

function isTwoWordName(s: string): boolean {
  const parts = s.trim().split(/\s+/);
  return parts.length >= 2 && parts.every((p) => p.length >= 2);
}

function checkDate(v: unknown, kind: 'past' | 'future'): ValidationOutcome {
  if (typeof v !== 'string' || !DATE_RE.test(v)) return { ok: false, reason: 'Expected ISO date YYYY-MM-DD' };
  const d = new Date(`${v}T00:00:00Z`).getTime();
  const now = Date.now();
  if (kind === 'past' && d >= now) return { ok: false, reason: 'Date must be in the past' };
  if (kind === 'future' && d <= now) return { ok: false, reason: 'Date must be in the future' };
  return { ok: true };
}

function checkHour(v: unknown, kind: 'past' | 'future'): ValidationOutcome {
  if (typeof v !== 'string' || !HOUR_RE.test(v)) return { ok: false, reason: 'Expected HH:mm' };
  const now = new Date();
  const [h, m] = v.split(':').map((p) => parseInt(p, 10));
  const candidate = new Date(now);
  candidate.setHours(h, m, 0, 0);
  if (kind === 'past' && candidate.getTime() >= now.getTime()) return { ok: false, reason: 'Hour must be in the past' };
  if (kind === 'future' && candidate.getTime() <= now.getTime()) return { ok: false, reason: 'Hour must be in the future' };
  return { ok: true };
}

function checkLength(v: unknown, rule: { min?: number; max?: number; exact?: number }): ValidationOutcome {
  const size = typeof v === 'string' ? v.length : typeof v === 'number' ? v : null;
  if (size === null) return { ok: false, reason: 'Length/range applies to strings and numbers only' };
  if (rule.exact !== undefined && size !== rule.exact) return { ok: false, reason: `Expected exactly ${String(rule.exact)}` };
  if (rule.min !== undefined && size < rule.min) return { ok: false, reason: `Expected at least ${String(rule.min)}` };
  if (rule.max !== undefined && size > rule.max) return { ok: false, reason: `Expected at most ${String(rule.max)}` };
  return { ok: true };
}
