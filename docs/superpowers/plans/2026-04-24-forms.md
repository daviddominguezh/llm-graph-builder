# Forms Implementation Plan (rev 2)

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans. Steps use checkbox (`- [ ]`) for tracking.

**Goal:** Ship the "forms" feature — agent-scoped structured data collection with tools, Data-tab configuration UI, right-panel inspection, streaming CSV export, and schema-coupling warnings.

**Architecture:** A `graph_forms` table per-agent (UUID PK, `display_name` + `form_slug` pair, FK to `graph_output_schemas`). Form data stored under `conversations.metadata.forms[<formUuid>]`. Two LLM tools (`set_form_fields`, `get_form_field`) registered under `OpenFlow/Forms`, backed by pure lib in `packages/api/src/lib/forms/` reused by right-panel display, CSV Route Handler, and unit tests. UI with shadcn/ui, keyboard-first.

**Tech:** TypeScript strict ESM, Postgres/Supabase (JSONB, RLS, SECURITY DEFINER RPCs), Next.js 16 (Server Actions + Route Handlers), Jest, shadcn/ui, next-intl.

**Spec:** [`docs/superpowers/specs/2026-04-24-forms-design.md`](../specs/2026-04-24-forms-design.md)

**Validation race note.** The spec's concern about "validate under lock" applies only to rules whose truth depends on current DB state. The MVP's validation rules (`email`, `twoWordName`, `pastDate`/`futureDate`, `pastHour`/`futureHour`, `length` on scalar fields or individual array-item values) are all **stateless** — they depend only on the input value, not on sibling data or array sizes. A concurrent writer cannot make a validated input become invalid. The validate-then-write pattern is therefore race-free for MVP. If future rules become stateful (e.g. "array length ≤ 5 across writes"), validation must move into SQL. This note is surfaced in the spec's acknowledged trade-offs.

**Testing:**
- `packages/api`: co-located `*.test.ts`, Jest ESM. `npm run test -w packages/api`.
- `packages/web`: no UI test harness — `npm run check` + manual smoke against `npm run dev -w packages/web` (port 3101).
- Pure-lib functions get real unit tests. SQL side is smoke-tested manually; add an integration test task at the end.

**Commit discipline:** TDD for pure lib (failing test → impl → passing → commit). For UI/DB, `npm run check` + commit per task. Stage files explicitly — never `git add .`, never `-am`, never `--no-verify`.

**File-size discipline:** every new file ≤300 lines, every function ≤40 lines. If a task's code threatens this, extract a helper file listed in the structure below.

```sql
-- supabase/migrations/20260424000000_graph_forms_table.sql
CREATE TABLE public.graph_forms (
  id           uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  agent_id     uuid NOT NULL REFERENCES public.agents(id) ON DELETE CASCADE,
  display_name text NOT NULL,
  form_slug    text NOT NULL,
  schema_id    text NOT NULL,
  validations  jsonb NOT NULL DEFAULT '{}'::jsonb,
  created_at   timestamptz NOT NULL DEFAULT now(),
  updated_at   timestamptz NOT NULL DEFAULT now(),
  CONSTRAINT graph_forms_agent_slug_unique UNIQUE (agent_id, form_slug),
  FOREIGN KEY (agent_id, schema_id)
    REFERENCES public.graph_output_schemas(agent_id, schema_id)
    ON DELETE RESTRICT,
  CONSTRAINT form_slug_format
    CHECK (form_slug ~ '^[a-z0-9]+(-[a-z0-9]+)*$'),
  CONSTRAINT form_slug_length
    CHECK (char_length(form_slug) BETWEEN 1 AND 64),
  CONSTRAINT form_slug_not_reserved
    CHECK (form_slug NOT IN ('new','all','any','none','edit','delete','create',
                             'export','import','settings','admin','api','null','undefined')),
  CONSTRAINT display_name_length
    CHECK (char_length(display_name) BETWEEN 1 AND 120)
);

CREATE INDEX graph_forms_schema_idx ON public.graph_forms (agent_id, schema_id);

CREATE TRIGGER graph_forms_set_updated_at
  BEFORE UPDATE ON public.graph_forms
  FOR EACH ROW EXECUTE FUNCTION public.update_updated_at();

ALTER TABLE public.graph_forms ENABLE ROW LEVEL SECURITY;

-- RLS: mirror graph_output_schemas — org members of the agent's owner org can read/write.
-- Replace the subquery shape if the existing schema policy uses a different helper.
CREATE POLICY "graph_forms_select"
  ON public.graph_forms FOR SELECT
  USING (EXISTS (
    SELECT 1 FROM public.agents a
    WHERE a.id = graph_forms.agent_id
      AND a.user_id = auth.uid()
  ));

CREATE POLICY "graph_forms_insert"
  ON public.graph_forms FOR INSERT
  WITH CHECK (EXISTS (
    SELECT 1 FROM public.agents a
    WHERE a.id = graph_forms.agent_id
      AND a.user_id = auth.uid()
  ));

CREATE POLICY "graph_forms_update"
  ON public.graph_forms FOR UPDATE
  USING (EXISTS (
    SELECT 1 FROM public.agents a
    WHERE a.id = graph_forms.agent_id
      AND a.user_id = auth.uid()
  ));

CREATE POLICY "graph_forms_delete"
  ON public.graph_forms FOR DELETE
  USING (EXISTS (
    SELECT 1 FROM public.agents a
    WHERE a.id = graph_forms.agent_id
      AND a.user_id = auth.uid()
  ));
```

**IMPORTANT:** if `20260312000000_reusable_output_schemas.sql` joins `org_members` (or uses an `is_org_member(...)` helper) instead of checking `agents.user_id = auth.uid()`, copy that pattern verbatim for all four policies. `agents` is currently owned by a single `user_id` per the codebase, so the direct `user_id = auth.uid()` pattern is correct **only if** schemas use the same. Verify first, then ship.

### Task 0.4: Migration — form RPCs (atomic write, failure append, array-max)

**Files:** create `supabase/migrations/20260424000002_form_rpcs.sql`

- [ ] **Step 1: Write**

```sql
-- supabase/migrations/20260424000002_form_rpcs.sql

-- Merge pre-validated form data into conversations.metadata.forms[<form_id>]
-- under a row lock. Field-level JSONB merge (new keys overwrite, siblings preserved).
CREATE OR REPLACE FUNCTION public.write_form_data(
  p_conversation_id uuid,
  p_form_id uuid,
  p_new_fields jsonb
) RETURNS void
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public, pg_temp
AS $$
BEGIN
  PERFORM 1 FROM public.conversations WHERE id = p_conversation_id FOR UPDATE;
  UPDATE public.conversations
     SET metadata = jsonb_set(
           COALESCE(metadata, '{}'::jsonb),
           ARRAY['forms', p_form_id::text],
           COALESCE(metadata->'forms'->p_form_id::text, '{}'::jsonb) || p_new_fields,
           true
         )
   WHERE id = p_conversation_id;
END;
$$;

-- Prepend a failure record and truncate to 3 newest entries in a single UPDATE.
CREATE OR REPLACE FUNCTION public.append_form_failure(
  p_conversation_id uuid,
  p_form_id uuid,
  p_entry jsonb
) RETURNS void
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public, pg_temp
AS $$
DECLARE
  _path text[] := ARRAY['forms_diagnostics', p_form_id::text, 'lastFailures'];
  _existing jsonb;
  _new_list jsonb;
BEGIN
  PERFORM 1 FROM public.conversations WHERE id = p_conversation_id FOR UPDATE;
  SELECT COALESCE(metadata #> _path, '[]'::jsonb)
    INTO _existing
    FROM public.conversations WHERE id = p_conversation_id;
  -- Prepend newest, keep first 3.
  SELECT COALESCE(jsonb_agg(elt), '[]'::jsonb)
    INTO _new_list
    FROM (
      SELECT elt FROM jsonb_array_elements(jsonb_build_array(p_entry) || _existing)
      WITH ORDINALITY AS t(elt, idx)
      ORDER BY idx ASC LIMIT 3
    ) s;
  UPDATE public.conversations
     SET metadata = jsonb_set(COALESCE(metadata, '{}'::jsonb), _path, _new_list, true)
   WHERE id = p_conversation_id;
END;
$$;

-- Observe max array length at a canonical container path (top-level arrays only for MVP).
-- For paths with multiple [] (multi-dim), returns the outer array's length only.
CREATE OR REPLACE FUNCTION public.form_array_max_length(
  p_agent uuid,
  p_tenant uuid,
  p_from timestamptz,
  p_to timestamptz,
  p_form_id uuid,
  p_container text
) RETURNS int
LANGUAGE sql STABLE
SET search_path = public, pg_temp
AS $$
  SELECT COALESCE(
    MAX(jsonb_array_length(metadata->'forms'->p_form_id::text->p_container)),
    0
  )
  FROM public.conversations
  WHERE agent_id = p_agent AND tenant_id = p_tenant
    AND created_at >= p_from AND created_at < p_to
    AND jsonb_typeof(metadata->'forms'->p_form_id::text->p_container) = 'array';
$$;

-- Grant: SECURITY DEFINER functions are callable by authenticated users via PostgREST.
-- Caller auth is enforced by Route Handler / Server Action before invoking.
GRANT EXECUTE ON FUNCTION public.write_form_data TO authenticated;
GRANT EXECUTE ON FUNCTION public.append_form_failure TO authenticated;
GRANT EXECUTE ON FUNCTION public.form_array_max_length TO authenticated;
```

```ts
const MAX_LEN = 64;

export function slugNormalize(input: string): string {
  const lower = input.toLowerCase();
  const hyphenated = lower.replace(/[\s_]+/g, '-');
```

```ts
// parsePath.ts
import type { PathSegment } from '../../types/forms.js';

export type ParseResult =
  | { ok: true; segments: PathSegment[] }
  | { ok: false; error: { reason: string; at: number } };

const FIELD_RE = /^[a-zA-Z_][a-zA-Z0-9_]*$/;

export function parsePath(input: string): ParseResult {
  if (input.length === 0) return fail('Empty path', 0);
  const rawSegments = input.split('.');
  const segments: PathSegment[] = [];
  let cursor = 0;
  for (const raw of rawSegments) {
    if (raw.length === 0) return fail('Empty segment', cursor);
    const r = parseSegment(raw, cursor);
    if (!r.ok) return r;
    segments.push(r.segment);
    cursor += raw.length + 1;
  }
  return { ok: true, segments };
}

type SegmentResult = { ok: true; segment: PathSegment } | Extract<ParseResult, { ok: false }>;

function parseSegment(raw: string, base: number): SegmentResult {
  const bracket = raw.indexOf('[');
  const name = bracket === -1 ? raw : raw.slice(0, bracket);
  if (!FIELD_RE.test(name)) return fail(`Invalid field name "${name}"`, base);
  const indicesPart = bracket === -1 ? '' : raw.slice(bracket);
  const idx = parseIndices(indicesPart, base + (bracket === -1 ? 0 : bracket));
  if (!idx.ok) return idx;
  return { ok: true, segment: { fieldName: name, indices: idx.list } };
}

type IdxResult = { ok: true; list: Array<number | 'wildcard'> } | Extract<ParseResult, { ok: false }>;

function parseIndices(s: string, cursor: number): IdxResult {
  const list: Array<number | 'wildcard'> = [];
  let i = 0;
  while (i < s.length) {
    if (s[i] !== '[') return fail(`Expected '[' at ${String(cursor + i)}`, cursor + i);
    const close = s.indexOf(']', i);
    if (close === -1) return fail('Unclosed bracket', cursor + i);
    const tok = s.slice(i + 1, close);
    if (tok === '') list.push('wildcard');
    else if (/^[0-9]+$/.test(tok)) list.push(parseInt(tok, 10));
    else return fail(`Invalid index "${tok}"`, cursor + i + 1);
    i = close + 1;
  }
  return { ok: true, list };
}

function fail(reason: string, at: number): Extract<ParseResult, { ok: false }> {
  return { ok: false, error: { reason, at } };
}
```

```ts
// zodForFieldPath.ts
import type { OutputSchemaField } from '@daviddh/graph-types';
import { z, type ZodTypeAny } from 'zod';
import { parsePath } from './parsePath.js';

export type ZodLookup =
  | { ok: true; zod: ZodTypeAny; expectedType: string }
  | { ok: false; reason: 'path-not-found' | 'parse-error'; detail?: string };

export function zodForFieldPath(fields: OutputSchemaField[], path: string): ZodLookup {
  const parsed = parsePath(path);
  if (!parsed.ok) return { ok: false, reason: 'parse-error', detail: parsed.error.reason };
  const field = resolve(fields, parsed.segments);
  if (!field) return { ok: false, reason: 'path-not-found' };
  return { ok: true, zod: toZod(field), expectedType: describe(field) };
}

function resolve(fields: OutputSchemaField[], segs: ReturnType<typeof parsePath> extends { ok: true; segments: infer S } ? S : never): OutputSchemaField | undefined {
  let pool: OutputSchemaField[] | undefined = fields;
  let cur: OutputSchemaField | undefined;
  for (const seg of segs) {
    if (!pool) return undefined;
    cur = pool.find((f) => f.name === seg.fieldName);
    if (!cur) return undefined;
    let walker = cur;
    for (let i = 0; i < seg.indices.length; i += 1) {
      if (walker.type !== 'array' || !walker.items) return undefined;
      walker = walker.items;
    }
    cur = walker;
    pool = cur.type === 'object' ? cur.properties : undefined;
  }
  return cur;
}

function toZod(f: OutputSchemaField): ZodTypeAny {
  switch (f.type) {
    case 'string': return z.string();
    case 'number': return z.number();
    case 'boolean': return z.boolean();
    case 'enum': return z.enum((f.enumValues ?? ['']) as [string, ...string[]]);
    case 'array': return z.array(f.items ? toZod(f.items) : z.unknown());
    case 'object': {
      const shape: Record<string, ZodTypeAny> = {};
      for (const p of f.properties ?? []) shape[p.name] = toZod(p);
      return z.object(shape);
    }
  }
}

function describe(f: OutputSchemaField): string {
  if (f.type === 'enum') return `enum(${(f.enumValues ?? []).join('|')})`;
  if (f.type === 'array' && f.items) return `array of ${describe(f.items)}`;
  return f.type;
}
```

```ts
// applyFormFields.ts
import type { ApplyResult, FieldApplyResult, FormData, FormDefinition } from '../../types/forms.js';
import { normalizePath } from './normalizePath.js';
import { parsePath } from './parsePath.js';
import { runValidation } from './runValidation.js';
import { zodForFieldPath } from './zodForFieldPath.js';

export interface ApplyInput {
  form: FormDefinition;
  currentData: FormData | undefined;
  fields: Array<{ fieldPath: string; fieldValue: unknown }>;
}

export function applyFormFields(input: ApplyInput): ApplyResult {
  const typeResults = typePass(input);
  if (!typeResults.every(isApplied)) return halt(typeResults, input);

  const ruleResults = rulePass(input);
  if (!ruleResults.every(isApplied)) return halt(ruleResults, input);

  const merged = mergeAll(input.currentData ?? {}, input.fields);
  if (!merged.ok) return { ok: false, newData: input.currentData ?? {}, results: merged.results };
  return { ok: true, newData: merged.data, results: ruleResults };
}

const isApplied = (r: FieldApplyResult): boolean => r.status === 'applied';

function halt(results: FieldApplyResult[], input: ApplyInput): ApplyResult {
  return { ok: false, newData: { ...(input.currentData ?? {}) }, results };
}

function typePass(input: ApplyInput): FieldApplyResult[] {
  return input.fields.map(({ fieldPath, fieldValue }) => {
    const p = parsePath(fieldPath);
    if (!p.ok) return { fieldPath, status: 'pathError', reason: p.error.reason };
    const look = zodForFieldPath(input.form.schemaFields, fieldPath);
    if (!look.ok) return { fieldPath, status: 'pathError', reason: 'Path does not exist on this form' };
    const v = look.zod.safeParse(fieldValue);
    if (!v.success) return { fieldPath, status: 'typeError', reason: v.error.issues[0]?.message ?? 'Type mismatch', expectedType: look.expectedType };
    return { fieldPath, status: 'applied', expectedType: look.expectedType };
  });
}

function rulePass(input: ApplyInput): FieldApplyResult[] {
  return input.fields.map(({ fieldPath, fieldValue }) => {
    const canonical = normalizePath(fieldPath);
    if (canonical === null) return { fieldPath, status: 'pathError', reason: 'Invalid path' };
    const rule = input.form.validations[canonical];
    if (!rule) return { fieldPath, status: 'applied' };
    const outcome = runValidation(fieldValue, rule);
    return outcome.ok
      ? { fieldPath, status: 'applied' }
      : { fieldPath, status: 'validationError', reason: outcome.reason };
  });
}

type MergeResult = { ok: true; data: FormData } | { ok: false; results: FieldApplyResult[] };

function mergeAll(base: FormData, fields: ApplyInput['fields']): MergeResult {
  const out: FormData = structuredClone(base);
  const errors: FieldApplyResult[] = [];
  for (const { fieldPath, fieldValue } of fields) {
    const r = setAtPath(out, fieldPath, fieldValue);
    if (!r.ok) errors.push({ fieldPath, status: 'pathError', reason: r.reason });
  }
  return errors.length === 0 ? { ok: true, data: out } : { ok: false, results: errors };
}

type SetResult = { ok: true } | { ok: false; reason: string };

function setAtPath(root: FormData, path: string, value: unknown): SetResult {
  const p = parsePath(path);
  if (!p.ok) return { ok: false, reason: p.error.reason };
  let parent: Record<string, unknown> | unknown[] = root;
  for (let i = 0; i < p.segments.length; i += 1) {
    const seg = p.segments[i];
    if (!seg) return { ok: false, reason: 'empty segment' };
    const isLastSegment = i === p.segments.length - 1;
    const r = stepInto(parent, seg, isLastSegment, value);
    if (!r.ok) return r;
    if (r.done) return { ok: true };
    parent = r.next;
  }
  return { ok: true };
}

interface StepDone { ok: true; done: true }
interface StepContinue { ok: true; done: false; next: Record<string, unknown> | unknown[] }
type StepResult = StepDone | StepContinue | { ok: false; reason: string };

function stepInto(
  parent: Record<string, unknown> | unknown[],
  seg: PathSegmentLike,
  isLastSegment: boolean,
  value: unknown
): StepResult {
  const container = getOrCreate(parent, seg.fieldName, seg.indices.length > 0 ? 'array' : 'object');
  if (!container.ok) return container;
  let ptr: unknown = container.value;
  for (let k = 0; k < seg.indices.length; k += 1) {
    const idx = seg.indices[k];
    if (idx === 'wildcard' || idx === undefined) return { ok: false, reason: 'Runtime path cannot contain wildcards' };
    if (!Array.isArray(ptr)) return { ok: false, reason: 'Expected array at index position' };
    const isLastIdx = k === seg.indices.length - 1;
    const mustSetValueHere = isLastSegment && isLastIdx;
    const kind: 'array' | 'object' = k < seg.indices.length - 1 ? 'array' : 'object';
    if (ptr[idx] === undefined) ptr[idx] = mustSetValueHere ? value : emptyOf(kind);
    if (mustSetValueHere) { ptr[idx] = value; return { ok: true, done: true }; }
    ptr = ptr[idx];
  }
  if (isLastSegment && seg.indices.length === 0) {
    (parent as Record<string, unknown>)[seg.fieldName] = value;
    return { ok: true, done: true };
  }
  if (!isObjectOrArray(ptr)) return { ok: false, reason: 'Expected object/array at path' };
  return { ok: true, done: false, next: ptr };
}

interface PathSegmentLike { fieldName: string; indices: Array<number | 'wildcard'> }

function getOrCreate(
  parent: Record<string, unknown> | unknown[],
  key: string,
  kind: 'array' | 'object'
): { ok: true; value: Record<string, unknown> | unknown[] } | { ok: false; reason: string } {
  if (Array.isArray(parent)) return { ok: false, reason: 'Cannot use field name on array parent' };
  if (parent[key] === undefined) { parent[key] = emptyOf(kind); return { ok: true, value: parent[key] as Record<string, unknown> | unknown[] }; }
  if (kind === 'array' && !Array.isArray(parent[key])) return { ok: false, reason: 'Existing value is not an array' };
```

```ts
// expandArrayColumns.ts
export const ARRAY_EXPANSION_CAP = 50;

export interface ExpandResult { columns: string[]; truncated: boolean }

export function expandArrayColumns(paths: string[], observedMax: Record<string, number>): ExpandResult {
  const columns: string[] = [];
  let truncated = false;
  for (const p of paths) {
    const slots = countSlots(p);
    if (slots === 0) { columns.push(p); continue; }
    const obs = observedMax[p] ?? 0;
    const budget = ARRAY_EXPANSION_CAP;
    const perms = enumeratePermutations(obs, slots, budget);
    for (const indices of perms) columns.push(subst(p, indices));
    if (perms.length === 0 || obs > perms.length) {
      truncated = true;
      columns.push(subst(p, new Array(slots).fill(`${String(budget)}+`)));
    }
  }
  return { columns, truncated };
}

function countSlots(p: string): number {
  return (p.match(/\[\]/g) ?? []).length;
}

function subst(p: string, idx: Array<number | string>): string {
  let i = 0;
  return p.replace(/\[\]/g, () => `[${String(idx[i++])}]`);
}

function enumeratePermutations(max: number, depth: number, budget: number): number[][] {
  if (max === 0 || depth === 0 || budget === 0) return depth === 0 ? [[]] : [];
  const out: number[][] = [];
  const perDim = Math.ceil(Math.pow(budget, 1 / depth));
  const cap = Math.min(max, perDim);
  for (let i = 0; i < cap && out.length < budget; i += 1) {
    for (const rest of enumeratePermutations(cap, depth - 1, budget - out.length)) {
      out.push([i, ...rest]);
      if (out.length >= budget) break;
    }
  }
  return out.slice(0, budget);
}
```

```ts
  conversationId: string;
  form: FormDefinition;
  fields: Array<{ fieldPath: string; fieldValue: unknown }>;
}

export async function applyFormFieldsAtomicQuery(args: ApplyArgs): Promise<ApplyResult> {
  const db = await createServerClient();
  const current = await queryFormData(args.conversationId, args.form.id);
  const result = applyFormFields({ form: args.form, currentData: current, fields: args.fields });
  if (!result.ok) return result;
  // Compute patch: changed keys only.
  const patch = topLevelDiff(current ?? {}, result.newData);

    p_conversation_id: args.conversationId,
    p_form_id: args.form.id,

  });

  return result;
}

function topLevelDiff(prev: Record<string, unknown>, next: Record<string, unknown>): Record<string, unknown> {
  const out: Record<string, unknown> = {};
  for (const k of Object.keys(next)) {
    const a = JSON.stringify(prev[k] ?? null);
    const b = JSON.stringify(next[k] ?? null);
    if (a !== b) out[k] = next[k];
  }
  return out;
}
```

```ts
export type AuthOutcome =
  | { ok: true; userId: string; formId: string }
  | { ok: false; status: 401 | 403 | 404; reason: string };

export async function authorizeFormAccess(args: {
  agentId: string;
  formSlug: string;
  tenantId: string;
}): Promise<AuthOutcome> {
  const db = await createServerClient();
  const { data: { user } } = await db.auth.getUser();
  if (!user) return { ok: false, status: 401, reason: 'unauthenticated' };

  // Verify caller owns the agent (matches existing agents-table RLS).
  const { data: agent } = await db.from('agents').select('id, user_id').eq('id', args.agentId).single();
  if (!agent) return { ok: false, status: 404, reason: 'agent-not-found' };
  if (agent.user_id !== user.id) return { ok: false, status: 403, reason: 'not-agent-owner' };

  // Verify caller is an org member of the tenant's org.
  const { data: tenant } = await db.from('tenants').select('id, org_id').eq('id', args.tenantId).single();
  if (!tenant) return { ok: false, status: 404, reason: 'tenant-not-found' };
  const { data: membership } = await db.from('org_members').select('user_id').eq('org_id', tenant.org_id).eq('user_id', user.id).maybeSingle();
  if (!membership) return { ok: false, status: 403, reason: 'not-org-member' };

  // Resolve slug → UUID.
  const { data: form } = await db.from('graph_forms').select('id').eq('agent_id', args.agentId).eq('form_slug', args.formSlug).single();
  if (!form) return { ok: false, status: 404, reason: 'form-not-found' };

  return { ok: true, userId: user.id, formId: form.id };
```

```ts
'use server';

import { revalidatePath } from 'next/cache';
import { slugNormalize, type ValidationsMap } from '@daviddh/llm-graph-runner';

import { createServerClient } from '@/lib/supabase/server';

export interface CreateFormInput {
  agentId: string;
  displayName: string;
  slug: string;
  schemaId: string;
  validations: ValidationsMap;
}

export async function createFormAction(input: CreateFormInput):
  Promise<{ ok: true; id: string } | { ok: false; reason: string }> {
  const slug = slugNormalize(input.slug);
  if (slug === '') return { ok: false, reason: 'invalid-slug' };
  if (input.displayName.trim() === '') return { ok: false, reason: 'invalid-name' };
  const db = await createServerClient();
  const { data, error } = await db.from('graph_forms').insert({
    agent_id: input.agentId, display_name: input.displayName.trim(),
    form_slug: slug, schema_id: input.schemaId, validations: input.validations,
  }).select('id').single();
  if (error) return { ok: false, reason: error.message };
  revalidatePath('/orgs');
  return { ok: true, id: data.id };
}

export async function updateFormValidationsAction(formId: string, validations: ValidationsMap):
  Promise<{ ok: true } | { ok: false; reason: string }> {
  const db = await createServerClient();
  const { error } = await db.from('graph_forms').update({ validations }).eq('id', formId);
  if (error) return { ok: false, reason: error.message };
  revalidatePath('/orgs');
  return { ok: true };
}

export async function deleteFormAction(formId: string):
  Promise<{ ok: true } | { ok: false; reason: string }> {
  const db = await createServerClient();
  const { error } = await db.from('graph_forms').delete().eq('id', formId);
  if (error) return { ok: false, reason: error.message };
  revalidatePath('/orgs');
  return { ok: true };
}

export async function checkSlugUniqueAction(agentId: string, slug: string): Promise<{ unique: boolean }> {
  const normalized = slugNormalize(slug);
  if (normalized === '') return { unique: false };
  const db = await createServerClient();
  const { count, error } = await db.from('graph_forms')
    .select('id', { count: 'exact', head: true })
    .eq('agent_id', agentId).eq('form_slug', normalized);
  if (error) throw error;
  return { unique: (count ?? 0) === 0 };
}

export async function listFormsAction(agentId: string):
  Promise<Array<{ id: string; slug: string; displayName: string; schemaId: string }>> {
  const db = await createServerClient();
  const { data, error } = await db.from('graph_forms')
    .select('id, form_slug, display_name, schema_id').eq('agent_id', agentId).order('created_at', { ascending: true });
  if (error) throw error;
  return (data ?? []).map((r) => ({ id: r.id, slug: r.form_slug, displayName: r.display_name, schemaId: r.schema_id }));
}

export async function getFormAction(formId: string):

  const { data, error } = await db.from('graph_forms').select('display_name, form_slug, schema_id, validations').eq('id', formId).single();

  return { displayName: data.display_name, slug: data.form_slug, schemaId: data.schema_id, validations: (data.validations ?? {}) as ValidationsMap };

}
```

```ts
export type SlugPhase = 'idle' | 'invalid-format' | 'checking' | 'unique' | 'taken-pending' | 'taken';

export interface SlugUxState { phase: SlugPhase; slug: string }

export type SlugUxAction =
  | { type: 'INPUT_CHANGED'; slug: string }
  | { type: 'UNIQ_RESULT'; unique: boolean }
  | { type: 'BLUR' };

export const initialSlugUx = (): SlugUxState => ({ phase: 'idle', slug: '' });

const FORMAT_RE = /^[a-z0-9]+(-[a-z0-9]+)*$/;

export function slugUxReducer(s: SlugUxState, a: SlugUxAction): SlugUxState {
  switch (a.type) {
    case 'INPUT_CHANGED':
      if (a.slug === '') return { phase: 'idle', slug: '' };
      if (!FORMAT_RE.test(a.slug)) return { phase: 'invalid-format', slug: a.slug };
      return { phase: 'checking', slug: a.slug };
    case 'UNIQ_RESULT':
      return { phase: a.unique ? 'unique' : 'taken-pending', slug: s.slug };
    case 'BLUR':
      return s.phase === 'taken-pending' ? { ...s, phase: 'taken' } : s;
  }
}
```

```tsx
const CHECK_DELAY_MS = 300;
const SPINNER_MIN_VISIBLE_MS = 200;

export function FormDialogNameSlug({ agentId, value, onChange, disabled }: Props): JSX.Element {
  const t = useTranslations('forms.field');
  const [slugEditing, setSlugEditing] = useState(false);
  const [state, dispatch] = useReducer(slugUxReducer, undefined, initialSlugUx);

    dispatch({ type: 'INPUT_CHANGED', slug: value.slug });

    const sinceLastKey = Date.now() - lastKeyRef.current;
    if (sinceLastKey < 800) setTimeout(() => dispatch({ type: 'BLUR' }), 800 - sinceLastKey);
    else dispatch({ type: 'BLUR' });
  }, []);

              value={value.slug} className="h-8"

            />
          ) : (
            <code className="rounded-md bg-muted px-2 py-1 text-xs" aria-label={value.slug || '—'}>{value.slug || '—'}</code>
          )}
          {!disabled && (
```

```tsx
export function FormDialogValidationsEditor({ schema, validations, onChange }: Props): JSX.Element {

  const [query, setQuery] = useState('');
  const [filter, setFilter] = useState<'all' | 'configured' | 'unconfigured'>('all');

  const passFilter = (path: string): boolean => {

    if (filter === 'configured' && validations[path] === undefined) return false;
    if (filter === 'unconfigured' && validations[path] !== undefined) return false;

  };

  const updateRule = (path: string, rule: ValidationRule | null): void => {
    const next = { ...validations };

      <p className="text-xs text-muted-foreground">{t('countProgress', { configured: configuredCount, total: leafPaths.length })}</p>

        <ToggleGroupItem value="all" size="sm">{t('filter.all')}</ToggleGroupItem>
        <ToggleGroupItem value="configured" size="sm">{t('filter.configured')}</ToggleGroupItem>
        <ToggleGroupItem value="unconfigured" size="sm">{t('filter.unconfigured')}</ToggleGroupItem>
      </ToggleGroup>
      <div className="flex flex-col">

      </div>

}

interface LeafInfo { path: string; type: LeafType }

function flattenLeafPaths(fields: OutputSchemaField[]): LeafInfo[] {
  const out: LeafInfo[] = [];
  const walk = (f: OutputSchemaField, p: string): void => {
    const path = p === '' ? f.name : `${p}.${f.name}`;
    if (f.type === 'object' && f.properties) { for (const pp of f.properties) walk(pp, path); return; }
    if (f.type === 'array' && f.items) { walkFromItems(f.items, `${path}[]`, out); return; }
    out.push({ path, type: f.type as LeafType });
  };
  const walkFromItems = (f: OutputSchemaField, p: string, acc: LeafInfo[]): void => {
    if (f.type === 'object' && f.properties) { for (const pp of f.properties) walk(pp, p === '' ? pp.name : `${p}.${pp.name}`); return; }
    if (f.type === 'array' && f.items) { walkFromItems(f.items, `${p}[]`, acc); return; }
    acc.push({ path: p, type: f.type as LeafType });
  };
  for (const f of fields) walk(f, '');
  return out;
}

function renderTree(
  fields: OutputSchemaField[],
  prefix: string,
  depth: number,
  leafPaths: LeafInfo[],
  validations: ValidationsMap,
  onChange: (path: string, rule: ValidationRule | null) => void,
  passFilter: (path: string) => boolean
): React.ReactNode[] {
  const out: React.ReactNode[] = [];
  for (const f of fields) {
    const thisPath = prefix === '' ? f.name : `${prefix}.${f.name}`;
    if (f.type === 'object' && f.properties) {
      const subPaths = leafPaths.filter((l) => l.path.startsWith(`${thisPath}.`));
      out.push(
        <FormValidationGroup key={thisPath} path={thisPath} kind="object" indent={depth}
          configuredCount={subPaths.filter((l) => validations[l.path] !== undefined).length}
          totalCount={subPaths.length}>
          {renderTree(f.properties, thisPath, depth + 1, leafPaths, validations, onChange, passFilter)}
        </FormValidationGroup>
      );
      continue;
    }
    if (f.type === 'array' && f.items) {
      const subPaths = leafPaths.filter((l) => l.path.startsWith(`${thisPath}[]`));
      out.push(
        <FormValidationGroup key={thisPath} path={thisPath} kind="array" indent={depth}
          configuredCount={subPaths.filter((l) => validations[l.path] !== undefined).length}
          totalCount={subPaths.length}>
          {renderTree([f.items], `${thisPath}[]`, depth + 1, leafPaths, validations, onChange, passFilter)}
        </FormValidationGroup>
      );
      continue;
    }
    // leaf
    if (!passFilter(thisPath)) continue;
    out.push(
      <FormValidationRow key={thisPath} path={thisPath} type={f.type as LeafType}
        rule={validations[thisPath] ?? null} indent={depth}
        onChange={(r) => onChange(thisPath, r)} />
    );
  }
  return out;
}
```

**Files:** create `packages/web/app/components/panels/FormDialog.tsx`

```tsx
'use client';

import type { OutputSchemaEntity } from '@daviddh/graph-types';
import { collectFieldPaths, type ValidationsMap } from '@daviddh/llm-graph-runner';
import { useEffect, useRef, useState } from 'react';
import { useTranslations } from 'next-intl';

import { Button } from '@/components/ui/button';
import { Dialog, DialogContent, DialogFooter, DialogHeader, DialogTitle } from '@/components/ui/dialog';
import { createFormAction, getFormAction, updateFormValidationsAction } from '@/app/actions/forms';

import { FormDialogStep1 } from './FormDialogStep1';
import { FormDialogStep2 } from './FormDialogStep2';

type Mode = { mode: 'create' } | { mode: 'edit'; formId: string };

interface Props {
  open: boolean; onClose: () => void;
  agentId: string; schemas: OutputSchemaEntity[]; mode: Mode;
}

interface Step1State { name: string; slug: string; schemaId: string | null; isNameSlugValid: boolean }

export function FormDialog({ open, onClose, agentId, schemas, mode }: Props): JSX.Element {
  const t = useTranslations('forms.dialog');
  const [step, setStep] = useState<1 | 2>(1);
  const [step1, setStep1] = useState<Step1State>({ name: '', slug: '', schemaId: null, isNameSlugValid: false });
  const [validations, setValidations] = useState<ValidationsMap>({});
  const [staleKept, setStaleKept] = useState(false);
  const [submitting, setSubmitting] = useState(false);
  const [dirty, setDirty] = useState(false);
  const contentRef = useRef<HTMLDivElement | null>(null);

  // Load form on edit open.
  useEffect(() => {
    if (!open || mode.mode !== 'edit') return;
    getFormAction(mode.formId).then((d) => {
      if (!d) return;
      setStep1({ name: d.displayName, slug: d.slug, schemaId: d.schemaId, isNameSlugValid: true });
      setValidations(d.validations);
    });
  }, [open, mode]);

  // Reset on close.
  useEffect(() => {
    if (open) return;
    setStep(1); setStep1({ name: '', slug: '', schemaId: null, isNameSlugValid: false });
    setValidations({}); setStaleKept(false); setSubmitting(false); setDirty(false);
  }, [open]);

  // Initial focus: Step 1 Name input; Step 2 first focusable.
  useEffect(() => {
    if (!open) return;
    const root = contentRef.current;
    if (!root) return;
    const target = step === 1 ? root.querySelector<HTMLInputElement>('#form-name') : root.querySelector<HTMLElement>('button, [role="combobox"], input');
    target?.focus();
  }, [open, step]);

  const selectedSchema = schemas.find((s) => s.id === step1.schemaId);
  const stalePaths = selectedSchema ? computeStale(selectedSchema.fields, validations) : [];

  const canNext = mode.mode === 'edit' ? true : step1.isNameSlugValid && step1.schemaId !== null;
  const canSubmit = stalePaths.length === 0 || staleKept;

  const submit = async (): Promise<void> => {
    if (!canSubmit) return;
    setSubmitting(true);
    if (mode.mode === 'create') {
      if (step1.schemaId === null) return;
      await createFormAction({ agentId, displayName: step1.name, slug: step1.slug, schemaId: step1.schemaId, validations });
    } else {
      await updateFormValidationsAction(mode.formId, validations);
    }
    setSubmitting(false);
    onClose();
  };

  const handleOpenChange = (next: boolean): void => {
    if (next) return;
    if (dirty && !window.confirm(t('unsavedConfirm'))) return;
    onClose();
  };

  const onKey = (e: React.KeyboardEvent): void => {
    if (e.key === 'Enter' && !e.shiftKey && !e.metaKey && !e.ctrlKey && step === 1 && canNext) {
      e.preventDefault(); setStep(2);
    } else if ((e.metaKey || e.ctrlKey) && e.key === 'Enter' && step === 2 && canSubmit) {
      e.preventDefault(); void submit();
    } else if (e.key === '/' && step === 2) {
      const input = contentRef.current?.querySelector<HTMLInputElement>('input[placeholder*="Search"]');
      if (input) { e.preventDefault(); input.focus(); }
    }
  };

  const titleKey = step === 1 ? `${mode.mode}.step1.title` : `${mode.mode}.step2.title`;

  return (
    <Dialog open={open} onOpenChange={handleOpenChange}>
      <DialogContent ref={contentRef} onKeyDown={onKey} className="max-w-xl" aria-labelledby="form-dialog-title">
        <DialogHeader>
          <DialogTitle id="form-dialog-title">{t(titleKey)}</DialogTitle>
          <p aria-live="polite" className="text-xs text-muted-foreground">{t('stepsCount', { step, total: 2 })}</p>
        </DialogHeader>
        {step === 1 ? (
          <FormDialogStep1 agentId={agentId} schemas={schemas} state={step1} disabled={mode.mode === 'edit'}
            onChange={(s) => { setStep1(s); setDirty(true); }} />
        ) : (
          <FormDialogStep2 schema={selectedSchema?.fields ?? []} validations={validations} stalePaths={stalePaths}
            staleKept={staleKept} onKeepStale={() => setStaleKept(true)}
            onChange={(v) => { setValidations(v); setDirty(true); }} />
        )}
        <DialogFooter>
          {step === 1 ? (
            <>
              <Button variant="ghost" onClick={() => handleOpenChange(false)}>{t('cancel')}</Button>
              <Button disabled={!canNext} onClick={() => setStep(2)}>{t('next')}</Button>
            </>
          ) : (
            <>
              <Button variant="ghost" onClick={() => setStep(1)}>{t('back')}</Button>
              <Button disabled={!canSubmit || submitting} onClick={submit}>{mode.mode === 'create' ? t('create') : t('save')}</Button>
            </>
          )}
        </DialogFooter>
      </DialogContent>
    </Dialog>
  );
}

function computeStale(fields: import('@daviddh/graph-types').OutputSchemaField[], validations: ValidationsMap): string[] {
  const canonical = new Set(collectFieldPaths(fields));
  return Object.keys(validations).filter((p) => !canonical.has(p));
}
```

```ts
import { NextRequest } from 'next/server';

import {
  ARRAY_EXPANSION_CAP, collectFieldPaths, expandArrayColumns, formatCsvRow, readFormField,
  type FormData,
} from '@daviddh/llm-graph-runner';
import type { OutputSchemaField } from '@daviddh/graph-types';

import { authorizeFormAccess } from '@/app/lib/forms/authorizeFormAccess';
import { createServerClient } from '@/lib/supabase/server';

const MAX_RANGE_DAYS = 15;
const PAGE_SIZE = 500;

export async function GET(req: NextRequest, ctx: { params: { agentId: string; formSlug: string } }): Promise<Response> {
  const sp = req.nextUrl.searchParams;
  const tenantId = sp.get('tenantId'); const from = sp.get('from'); const to = sp.get('to');
  const tenantSlug = sp.get('tenantSlug') ?? 'tenant';
  const agentSlug = sp.get('agentSlug') ?? 'agent';
  if (!tenantId || !from || !to) return new Response('missing-params', { status: 400 });
  if ((Date.parse(to) - Date.parse(from)) / 86_400_000 > MAX_RANGE_DAYS) return new Response('invalid-range', { status: 400 });

  const auth = await authorizeFormAccess({ agentId: ctx.params.agentId, formSlug: ctx.params.formSlug, tenantId });
  if (!auth.ok) return new Response(auth.reason, { status: auth.status });

  const db = await createServerClient();
  const { data: formRow } = await db.from('graph_forms').select('id, schema_id').eq('id', auth.formId).single();
  if (!formRow) return new Response('form-not-found', { status: 404 });
  const { data: schemaRow } = await db.from('graph_output_schemas').select('fields').eq('agent_id', ctx.params.agentId).eq('schema_id', formRow.schema_id).single();
  if (!schemaRow) return new Response('schema-not-found', { status: 404 });
  const schemaFields = schemaRow.fields as OutputSchemaField[];
  const canonicalPaths = collectFieldPaths(schemaFields);

  const observedMax = await computeObservedMax(db, ctx.params.agentId, tenantId, from, to, formRow.id, canonicalPaths);
  const { columns, truncated } = expandArrayColumns(canonicalPaths, observedMax);
  const fixed = ['conversation_id', 'user_name', 'channel', 'started_at', 'last_message_at', 'status'];
  const filename = buildFilename(tenantSlug, agentSlug, ctx.params.formSlug, from, to);

  const stream = new ReadableStream<Uint8Array>({
    async start(controller) {
      const encoder = new TextEncoder();
      controller.enqueue(encoder.encode('﻿' + formatCsvRow([...fixed, ...columns]) + '\n'));
      let cursor: { ts: string; id: string } | null = null;
      let rowsWritten = 0;
      for (;;) {
        const batch = await fetchBatch(db, { tenantId, agentId: ctx.params.agentId, formId: formRow.id, from, to, cursor });
        if (batch.length === 0) break;
        for (const row of batch) {
          const fd = row.metadata?.forms?.[formRow.id] as FormData | undefined;
          const dynCells = columns.map((c) => {
            const r = readFormField(fd, c);
            return r.ok && r.value !== undefined ? String(r.value) : '';
          });
          if (dynCells.every((c) => c === '')) continue;  // drop empties
          const cells = [row.id, row.user_name ?? '', row.channel, row.created_at, row.last_message_at ?? '', row.status, ...dynCells];
          controller.enqueue(encoder.encode(formatCsvRow(cells) + '\n'));
          rowsWritten += 1;
        }
        const last = batch[batch.length - 1]!;
        cursor = { ts: last.created_at, id: last.id };
        if (batch.length < PAGE_SIZE) break;
      }
      controller.close();
    },
  });

  return new Response(stream, {
    headers: {
      'content-type': 'text/csv; charset=utf-8',
      'x-forms-truncated': truncated ? 'true' : 'false',
      'content-disposition': `attachment; filename="${filename}"`,
    },
  });
}

function buildFilename(tenantSlug: string, agentSlug: string, formSlug: string, from: string, to: string): string {
  const base = `openflow-${tenantSlug}-${agentSlug}-${formSlug}-${from}-${to}.csv`;
  if (base.length <= 200) return base;
  const hash = Math.abs(hashString(base)).toString(16).slice(0, 8);
  return `${base.slice(0, 180)}-${hash}.csv`;
}

function hashString(s: string): number {
  let h = 0; for (let i = 0; i < s.length; i += 1) { h = ((h << 5) - h) + s.charCodeAt(i); h |= 0; } return h;
}

async function computeObservedMax(
  db: Awaited<ReturnType<typeof createServerClient>>,
  agentId: string, tenantId: string, from: string, to: string, formId: string, paths: string[]
): Promise<Record<string, number>> {
  const out: Record<string, number> = {};
  const arrayPaths = paths.filter((p) => p.includes('[]'));
  for (const p of arrayPaths) {
    const container = p.split('[')[0];  // everything before first bracket — outer array name
    const { data } = await db.rpc('form_array_max_length', {
      p_agent: agentId, p_tenant: tenantId, p_from: from, p_to: to, p_form_id: formId, p_container: container,
    });
    out[p] = Number(data ?? 0);
  }
  return out;
}

interface Row { id: string; user_name: string | null; channel: string; created_at: string; last_message_at: string | null; status: string; metadata: { forms?: Record<string, unknown> } | null }

async function fetchBatch(
  db: Awaited<ReturnType<typeof createServerClient>>,
  p: { tenantId: string; agentId: string; formId: string; from: string; to: string; cursor: { ts: string; id: string } | null }
): Promise<Row[]> {
  let q = db.from('conversations').select('id, channel, created_at, last_message_at, status, metadata')
    .eq('tenant_id', p.tenantId).eq('agent_id', p.agentId)
    .gte('created_at', p.from).lt('created_at', p.to)
    .not(`metadata->forms->${p.formId}`, 'is', null)
    .order('created_at').order('id').limit(PAGE_SIZE);
  if (p.cursor) {
    // Keyset with tiebreaker: (created_at > cursor.ts) OR (created_at = cursor.ts AND id > cursor.id)
    q = q.or(`created_at.gt.${p.cursor.ts},and(created_at.eq.${p.cursor.ts},id.gt.${p.cursor.id})`);
  }
  const { data } = await q;
  return ((data ?? []) as unknown) as Row[];
}
```

### Task 9.4: `useCsvExport` hook — AbortController + toast + truncated header

**Files:** create `packages/web/app/components/messages/components/ExportCsv/useCsvExport.ts`

```ts
'use client';
import { useRef, useState } from 'react';
import { toast } from 'sonner';  // or the project's toast system
import { useTranslations } from 'next-intl';

export type ExportState = 'idle' | 'generating';

export function useCsvExport(): {
  state: ExportState;
  run: (url: string, filename: string) => Promise<void>;
  abort: () => void;
} {
  const t = useTranslations('forms.export');
  const [state, setState] = useState<ExportState>('idle');
  const controllerRef = useRef<AbortController | null>(null);

  const run = async (url: string, filename: string): Promise<void> => {
    setState('generating');
    const ac = new AbortController();
    controllerRef.current = ac;
    try {
      const res = await fetch(url, { signal: ac.signal });
      if (!res.ok) throw new Error(`http-${res.status}`);
      const truncated = res.headers.get('x-forms-truncated') === 'true';
      const blob = await res.blob();
      const rows = countRows(await blob.text());
      if (rows === 0) { toast.error(t('errorRaceEmpty')); return; }
      const dl = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = dl; a.download = filename; a.click();
      URL.revokeObjectURL(dl);
      toast.success(truncated ? t('successTruncated', { count: rows }) : t('success', { count: rows }));
    } catch (err) {
      if ((err as Error).name === 'AbortError') return;
      toast.error(String(err));
    } finally {
      setState('idle');
      controllerRef.current = null;
    }
  };

  const abort = (): void => { controllerRef.current?.abort(); };
  return { state, run, abort };
}

function countRows(csv: string): number {
  const lines = csv.split('\n').filter((l) => l.length > 0);
  return Math.max(0, lines.length - 1);  // minus header
}
```
