# VFS Core Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the three-layer caching proxy (in-memory → Supabase Storage → source provider) with dirty-set coherence, path validation, tree indexing, and session tracking.

**Architecture:** Layered modules in `packages/api/src/vfs/`. Each layer is independently testable. `VFSContext` is the coordinator that orchestrates the read/write paths through these layers. A Supabase migration creates the `vfs_sessions` table, Storage bucket, and RLS policies. A `vfs-cleanup` Edge Function handles stale session garbage collection.

**Tech Stack:** TypeScript (ESM, strict mode), Jest (ESM preset), picomatch (glob matching), @upstash/redis (dirty set), @supabase/supabase-js (Storage + Postgres)

**Spec:** `docs/superpowers/specs/2026-03-30-vfs-core-design.md`

**ESLint constraints:** max 40 lines/function (skip blanks/comments), max 300 lines/file, max depth 2. When hitting limits, extract helpers into separate files.

```typescript
import picomatch from 'picomatch';

import { type PathValidationConfig, VFSError, VFSErrorCode } from './types.js';

const HARDCODED_BLOCKED: readonly string[] = ['.git/**', '.git'];
const DEFAULT_BLOCKED: readonly string[] = ['node_modules/**', '.env', '.env.*'];

function normalizePath(path: string): string {
  let result = path;
  if (result.startsWith('./')) result = result.slice(2);
  result = result.replace(/\/+/g, '/');
  if (result.endsWith('/')) result = result.slice(0, -1);
  return result;
}

function throwPathError(code: VFSErrorCode, message: string): never {
  throw new VFSError(code, message);
}

function validateStructure(path: string): string {
  if (path === '') throwPathError(VFSErrorCode.INVALID_PATH, 'Path cannot be empty');
  if (path.startsWith('/')) throwPathError(VFSErrorCode.INVALID_PATH, 'Path must be relative');
  if (path.includes('..')) throwPathError(VFSErrorCode.INVALID_PATH, 'Path traversal not allowed');
  if (path.includes('\0')) throwPathError(VFSErrorCode.INVALID_PATH, 'Path contains null bytes');
  return normalizePath(path);
}

function matchesAny(path: string, patterns: readonly string[]): boolean {
  return patterns.some((p) => picomatch(p)(path));
}

export function validatePath(path: string, _config?: PathValidationConfig): void {
  const normalized = validateStructure(path);
  if (matchesAny(normalized, HARDCODED_BLOCKED)) {
    throwPathError(VFSErrorCode.PERMISSION_DENIED, `Access to ${normalized} is blocked`);
  }
}

export function validateWritePath(path: string, config?: PathValidationConfig): void {
  validatePath(path, config);
  const normalized = normalizePath(path);
  const blocked = config?.blockedPatterns ?? DEFAULT_BLOCKED;
  if (matchesAny(normalized, blocked)) {
    throwPathError(VFSErrorCode.PERMISSION_DENIED, `Write to ${normalized} is blocked`);
  }
}
```

```typescript
import type { CachedFile } from './types.js';

export class MemoryLayer {
  private readonly files = new Map<string, CachedFile>();

  get(path: string): CachedFile | undefined {
    return this.files.get(path);
  }

  set(path: string, content: string, updatedAt: number): void {
    this.files.set(path, { content, updatedAt });
  }

  delete(path: string): boolean {
    return this.files.delete(path);
  }

  rename(oldPath: string, newPath: string): boolean {
    const file = this.files.get(oldPath);
    if (file === undefined) return false;
    this.files.delete(oldPath);
    this.files.set(newPath, file);
    return true;
  }

  has(path: string): boolean {
    return this.files.has(path);
  }

  paths(): string[] {
    return [...this.files.keys()];
  }

  entries(): IterableIterator<[string, CachedFile]> {
    return this.files.entries();
  }
}
```

```typescript
import type { RedisClient } from './types.js';

const TTL_SECONDS = 900;
const TREE_INDEX_KEY = '__tree_index';

export class DirtySetClient {
  private readonly redisKey: string;

  constructor(
    private readonly redis: RedisClient,
    sessionKey: string
  ) {
    this.redisKey = `vfs:dirty:${sessionKey}`;
  }

  async getTimestamp(path: string): Promise<number | null> {
    try {
      const value = await this.redis.hget(this.redisKey, path);
      return value !== null ? Number(value) : null;
    } catch {
      return Date.now();
    }
  }

  async getTimestamps(paths: string[]): Promise<Map<string, number>> {
    try {
      const values = await this.redis.hmget(this.redisKey, ...paths);
      return buildTimestampMap(paths, values);
    } catch {
      return buildFallbackMap(paths);
    }
  }

  async markDirty(path: string, timestamp: number): Promise<void> {
    try {
      const pipe = this.redis.pipeline();
      pipe.hset(this.redisKey, { [path]: String(timestamp) });
      pipe.expire(this.redisKey, TTL_SECONDS);
      await pipe.exec();
    } catch {
      // Redis is an optimization — silently no-op on failure
    }
  }

  async markTreeDirty(timestamp: number): Promise<void> {
    return this.markDirty(TREE_INDEX_KEY, timestamp);
  }

  async getTreeTimestamp(): Promise<number | null> {
    try {
      const value = await this.redis.hget(this.redisKey, TREE_INDEX_KEY);
      return value !== null ? Number(value) : null;
    } catch {
      return null;
    }
  }

  async deleteKey(): Promise<void> {
    try {
      await this.redis.del(this.redisKey);
    } catch {
      // Best-effort cleanup
    }
  }
}

function buildTimestampMap(paths: string[], values: Array<string | null>): Map<string, number> {
  const map = new Map<string, number>();
  for (let i = 0; i < paths.length; i++) {
    const val = values[i];
    if (val !== null && val !== undefined) {
      map.set(paths[i]!, Number(val));
    }
  }
  return map;
}

function buildFallbackMap(paths: string[]): Map<string, number> {
  const now = Date.now();
  const map = new Map<string, number>();
  for (const path of paths) {
    map.set(path, now);
  }
  return map;
}
```

```typescript
import type { TreeEntry, TreeNode } from './types.js';

export const DEFAULT_IGNORES = ['.git', 'node_modules', '__pycache__', '.next', 'dist', 'build'];

const LANGUAGE_MAP: Record<string, string> = {
  ts: 'typescript',
  tsx: 'typescript',
  js: 'javascript',
  jsx: 'javascript',
  py: 'python',
  go: 'go',
  rs: 'rust',
  java: 'java',
  rb: 'ruby',
  php: 'php',
  c: 'c',
  cpp: 'cpp',
  h: 'c',
  cs: 'csharp',
  swift: 'swift',
  kt: 'kotlin',
  json: 'json',
  yaml: 'yaml',
  yml: 'yaml',
  md: 'markdown',
  html: 'html',
  css: 'css',
  scss: 'scss',
  sql: 'sql',
  sh: 'shell',
  bash: 'shell',
  toml: 'toml',
  xml: 'xml',
  graphql: 'graphql',
};

export function inferLanguage(path: string): string {
  const ext = path.split('.').pop()?.toLowerCase() ?? '';
  return LANGUAGE_MAP[ext] ?? 'unknown';
}

export function shouldIgnore(path: string): boolean {
  const firstSegment = path.split('/')[0] ?? '';
  return DEFAULT_IGNORES.includes(firstSegment);
}

export function buildNestedTree(
  entries: Map<string, TreeEntry>,
  rootPath: string,
  maxDepth: number
): TreeNode {
  const name = rootPath === '' ? '.' : rootPath.split('/').pop() ?? rootPath;
  const node: TreeNode = { name, type: 'directory', path: rootPath };
  if (maxDepth <= 0) return node;
  node.children = buildChildren(entries, rootPath, maxDepth);
  return node;
}

function buildChildren(
  entries: Map<string, TreeEntry>,
  parentPath: string,
  maxDepth: number
): TreeNode[] {
  const children: TreeNode[] = [];
  const prefix = parentPath === '' ? '' : `${parentPath}/`;

  for (const [path, entry] of entries) {
    if (!isDirectChild(path, prefix)) continue;
    children.push(entryToNode(entries, entry, maxDepth));
  }

  return sortChildren(children);
}

function isDirectChild(path: string, prefix: string): boolean {
  if (!path.startsWith(prefix)) return false;
  const rest = path.slice(prefix.length);
  return !rest.includes('/');
}

function entryToNode(
  entries: Map<string, TreeEntry>,
  entry: TreeEntry,
  maxDepth: number
): TreeNode {
  if (entry.type === 'directory') {
    return buildNestedTree(entries, entry.path, maxDepth - 1);
  }
  return {
    name: entry.path.split('/').pop() ?? entry.path,
    type: 'file',
    path: entry.path,
    sizeBytes: entry.sizeBytes,
    language: inferLanguage(entry.path),
  };
}

function sortChildren(children: TreeNode[]): TreeNode[] {
  const dirs = children.filter((c) => c.type === 'directory').sort((a, b) => a.name.localeCompare(b.name));
  const files = children.filter((c) => c.type === 'file').sort((a, b) => a.name.localeCompare(b.name));
  return [...dirs, ...files];
}
```

```typescript
import picomatch from 'picomatch';

import type { TreeEntry, TreeNode } from './types.js';
import { buildNestedTree, inferLanguage, shouldIgnore } from './treeIndexHelpers.js';

export class TreeIndex {
  private tree: TreeNode | null = null;
  private flatEntries = new Map<string, TreeEntry>();
  private updatedAt: number | null = null;

  isLoaded(): boolean {
    return this.updatedAt !== null;
  }

  getUpdatedAt(): number | null {
    return this.updatedAt;
  }

  load(entries: TreeEntry[], updatedAt: number): void {
    this.flatEntries.clear();
    for (const entry of entries) {
      if (!shouldIgnore(entry.path)) this.flatEntries.set(entry.path, entry);
    }
    this.tree = buildNestedTree(this.flatEntries, '', 99);
    this.updatedAt = updatedAt;
  }

  exists(path: string): boolean {
    return this.flatEntries.has(path);
  }

  isDirectory(path: string): boolean {
    return this.flatEntries.get(path)?.type === 'directory';
  }

  getMetadata(path: string): { sizeBytes?: number; language?: string } | null {
    const entry = this.flatEntries.get(path);
    if (entry === undefined || entry.type !== 'file') return null;
    return { sizeBytes: entry.sizeBytes, language: inferLanguage(path) };
  }

  findFiles(pattern: string, path?: string, exclude?: string[]): string[] {
    const matcher = picomatch(pattern);
    const excludeMatcher = exclude?.length ? picomatch(exclude) : null;
    const results: string[] = [];
    for (const [filePath, entry] of this.flatEntries) {
      if (entry.type !== 'file') continue;
      if (path !== undefined && path !== '' && !filePath.startsWith(`${path}/`)) continue;
      if (excludeMatcher !== null && excludeMatcher(filePath)) continue;
      if (matcher(filePath)) results.push(filePath);
    }
    return results;
  }

  listDirectory(path: string, recursive?: boolean, maxDepth?: number): TreeEntry[] {
    const prefix = path === '' ? '' : `${path}/`;
    const depth = maxDepth ?? 2;
    const results: TreeEntry[] = [];
    for (const [filePath, entry] of this.flatEntries) {
      if (!isInScope(filePath, prefix, recursive ?? false, depth)) continue;
      results.push(entry);
    }
    return results;
  }

  getTree(path?: string, maxDepth?: number): TreeNode | null {
    if (this.tree === null) return null;
    if (path === undefined || path === '') {
      return buildNestedTree(this.flatEntries, '', maxDepth ?? 3);
    }
    return buildNestedTree(this.flatEntries, path, maxDepth ?? 3);
  }

  addFile(path: string, sizeBytes: number): void {
    this.flatEntries.set(path, { path, type: 'file', sizeBytes });
    this.ensureParentDirs(path);
    this.rebuildTree();
  }

  removeFile(path: string): void {
    this.flatEntries.delete(path);
    this.rebuildTree();
  }

  moveFile(oldPath: string, newPath: string): void {
    const entry = this.flatEntries.get(oldPath);
    if (entry === undefined) return;
    this.flatEntries.delete(oldPath);
    this.flatEntries.set(newPath, { ...entry, path: newPath });
    this.ensureParentDirs(newPath);
    this.rebuildTree();
  }

  updateFileSize(path: string, sizeBytes: number): void {
    const entry = this.flatEntries.get(path);
    if (entry !== undefined) entry.sizeBytes = sizeBytes;
  }

  serialize(): string {
    return JSON.stringify([...this.flatEntries.values()]);
  }

  static deserialize(data: string, updatedAt: number): TreeIndex {
    const entries: TreeEntry[] = JSON.parse(data) as TreeEntry[];
    const index = new TreeIndex();
    for (const entry of entries) {
      index.flatEntries.set(entry.path, entry);
    }
    index.tree = buildNestedTree(index.flatEntries, '', 99);
    index.updatedAt = updatedAt;
    return index;
  }

  private ensureParentDirs(path: string): void {
    const parts = path.split('/');
    for (let i = 1; i < parts.length; i++) {
      const dirPath = parts.slice(0, i).join('/');
      if (!this.flatEntries.has(dirPath)) {
        this.flatEntries.set(dirPath, { path: dirPath, type: 'directory' });
      }
    }
  }

  private rebuildTree(): void {
    this.tree = buildNestedTree(this.flatEntries, '', 99);
  }
}

function isInScope(filePath: string, prefix: string, recursive: boolean, maxDepth: number): boolean {
  if (prefix !== '' && !filePath.startsWith(prefix)) return false;
  if (!recursive) {
    const rest = filePath.slice(prefix.length);
    return !rest.includes('/');
  }
  const rest = filePath.slice(prefix.length);
  const depth = rest.split('/').length;
  return depth <= maxDepth;
}
```

```typescript
import { type StorageBucketApi, type SupabaseVFSClient, VFSError, VFSErrorCode } from './types.js';

const BUCKET = 'vfs';
const TREE_INDEX_FILE = '__tree_index.json';
const LIST_PAGE_SIZE = 100;

export class StorageLayer {
  private readonly bucket: StorageBucketApi;
  private readonly prefix: string;

  constructor(supabase: SupabaseVFSClient, sessionPrefix: string) {
    this.bucket = supabase.storage.from(BUCKET);
    this.prefix = sessionPrefix;
  }

  async upload(path: string, content: string): Promise<void> {
    const { error } = await this.bucket.upload(this.fullPath(path), content, {
      contentType: 'text/plain',
      upsert: true,
    });
    if (error !== null) throw wrapStorageError(error, 'upload', path);
  }

  async download(path: string): Promise<string | null> {
    const { data, error } = await this.bucket.download(this.fullPath(path));
    if (error !== null) {
      if (isNotFound(error)) return null;
      throw wrapStorageError(error, 'download', path);
    }
    return data !== null ? await data.text() : null;
  }

  async delete(path: string): Promise<void> {
    const { error } = await this.bucket.remove([this.fullPath(path)]);
    if (error !== null) throw wrapStorageError(error, 'delete', path);
  }

  async rename(oldPath: string, newPath: string): Promise<void> {
    // copy-then-delete — not atomic, documented in spec
    const { error: copyErr } = await this.bucket.copy(this.fullPath(oldPath), this.fullPath(newPath));
    if (copyErr !== null) throw wrapStorageError(copyErr, 'rename (copy)', oldPath);
    const { error: delErr } = await this.bucket.remove([this.fullPath(oldPath)]);
    if (delErr !== null) throw wrapStorageError(delErr, 'rename (delete)', oldPath);
  }

  async uploadTreeIndex(data: string): Promise<void> {
    return this.upload(TREE_INDEX_FILE, data);
  }

  async downloadTreeIndex(): Promise<string | null> {
    return this.download(TREE_INDEX_FILE);
  }

  async deleteAll(): Promise<void> {
    const allPaths = await this.listAllRecursive(this.prefix);
    if (allPaths.length === 0) return;
    for (let i = 0; i < allPaths.length; i += LIST_PAGE_SIZE) {
      const batch = allPaths.slice(i, i + LIST_PAGE_SIZE);
      const { error } = await this.bucket.remove(batch);
      if (error !== null) throw wrapStorageError(error, 'deleteAll', this.prefix);
    }
  }

  private fullPath(path: string): string {
    return `${this.prefix}/${path}`;
  }

  private async listAllRecursive(prefix: string): Promise<string[]> {
    const paths: string[] = [];
    let offset = 0;
    let hasMore = true;

    while (hasMore) {
      const { data, error } = await this.bucket.list(prefix, { limit: LIST_PAGE_SIZE, offset });
      if (error !== null) throw wrapStorageError(error, 'list', prefix);
      if (data === null || data.length === 0) break;
      for (const item of data) {
        const fullPath = `${prefix}/${item.name}`;
        if (item.id === null) {
          const nested = await this.listAllRecursive(fullPath);
          paths.push(...nested);
        } else {
          paths.push(fullPath);
        }
      }
      hasMore = data.length === LIST_PAGE_SIZE;
      offset += LIST_PAGE_SIZE;
    }

    return paths;
  }
}

function isNotFound(error: { message: string; statusCode?: string }): boolean {
  return error.statusCode === '404' || error.message.includes('not found');
}

function wrapStorageError(
  error: { message: string },
  operation: string,
  path: string
): VFSError {
  return new VFSError(VFSErrorCode.PROVIDER_ERROR, `Storage ${operation} failed for ${path}: ${error.message}`);
}
```

```typescript
const THROTTLE_MS = 60_000;
const TABLE = 'vfs_sessions';

interface InitializeParams {
  tenantSlug: string;
  agentSlug: string;
  userID: string;
  sessionId: string;
  commitSha: string;
}

export class SessionTracker {
  private lastTouchTime = 0;

  constructor(
    private readonly supabase: SupabaseVFSClient,
    private readonly sessionKey: string
  ) {}

  async initialize(params: InitializeParams): Promise<void> {
    const { error } = await this.supabase
      .from(TABLE)
      .upsert(
        {
          session_key: this.sessionKey,
          tenant_slug: params.tenantSlug,
          agent_slug: params.agentSlug,
          user_id: params.userID,
          session_id: params.sessionId,
          commit_sha: params.commitSha,
        },
        { onConflict: 'session_key' }
      )
      .select()
      .single();

    if (error !== null) throw new Error(`Failed to initialize VFS session: ${error.message}`);
    this.lastTouchTime = Date.now();
  }

  async touch(): Promise<void> {
    const now = Date.now();
    if (now - this.lastTouchTime < THROTTLE_MS) return;
    this.lastTouchTime = now;

    await this.supabase
      .from(TABLE)
      .update({ last_accessed_at: new Date().toISOString() })
      .eq('session_key', this.sessionKey);
  }
}
```

```typescript
import { Redis } from '@upstash/redis';
import { createClient } from '@supabase/supabase-js';

const BUCKET = 'vfs';
const LIST_PAGE_SIZE = 100;
const STALE_INTERVAL = "30 minutes";

interface StaleSession {
  session_key: string;
}

function authenticateRequest(req: Request): Response | null {
  const masterKey = Deno.env.get('EDGE_FUNCTION_MASTER_KEY');
  if (masterKey === undefined || masterKey === '') {
    return new Response('Server misconfigured', { status: 500 });
  }
  const token = req.headers.get('x-master-key') ?? '';
  if (token === '' || token !== masterKey) {
    return new Response('Unauthorized', { status: 401 });
  }
  return null;
}

async function listAllRecursive(
  bucket: ReturnType<ReturnType<typeof createClient>['storage']['from']>,
  prefix: string
): Promise<string[]> {
  const paths: string[] = [];
  let offset = 0;
  let hasMore = true;
  while (hasMore) {
    const { data, error } = await bucket.list(prefix, { limit: LIST_PAGE_SIZE, offset });
    if (error !== null || data === null || data.length === 0) break;
    for (const item of data) {
      const fullPath = `${prefix}/${item.name}`;
      if (item.id === null) {
        const nested = await listAllRecursive(bucket, fullPath);
        paths.push(...nested);
      } else {
        paths.push(fullPath);
      }
    }
    hasMore = data.length === LIST_PAGE_SIZE;
    offset += LIST_PAGE_SIZE;
  }
  return paths;
}

async function cleanupSession(
  supabase: ReturnType<typeof createClient>,
  redis: Redis,
  session: StaleSession
): Promise<void> {
  // 1. Delete Storage objects (first)
  const bucket = supabase.storage.from(BUCKET);
  const paths = await listAllRecursive(bucket, session.session_key);
  for (let i = 0; i < paths.length; i += LIST_PAGE_SIZE) {
    await bucket.remove(paths.slice(i, i + LIST_PAGE_SIZE));
  }

  // 2. Delete Redis dirty set (second)
  try {
    await redis.del(`vfs:dirty:${session.session_key}`);
  } catch {
    // Best-effort — key TTLs at 15 min anyway
  }

  // 3. Delete DB row (last — anchor for retry)
  await supabase.from('vfs_sessions').delete().eq('session_key', session.session_key);
}

Deno.serve(async (req: Request) => {
  if (req.method === 'OPTIONS') {
    return new Response('ok', { status: 200 });
  }

  const authError = authenticateRequest(req);
  if (authError !== null) return authError;

  const supabase = createClient(
    Deno.env.get('SUPABASE_URL')!,
    Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
  );
  const redis = new Redis({
    url: Deno.env.get('UPSTASH_REDIS_REST_URL')!,
    token: Deno.env.get('UPSTASH_REDIS_REST_TOKEN')!,
  });

  const { data: staleSessions } = await supabase
    .from('vfs_sessions')
    .select('session_key')
    .lt('last_accessed_at', new Date(Date.now() - 30 * 60 * 1000).toISOString());

  if (staleSessions !== null) {
    for (const session of staleSessions as StaleSession[]) {
      await cleanupSession(supabase, redis, session);
    }
  }

  return new Response(JSON.stringify({ cleaned: staleSessions?.length ?? 0 }), {
    headers: { 'Content-Type': 'application/json' },
  });
});
```
